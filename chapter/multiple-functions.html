দ্বিগুণ টাকা ধার করে পালালে

লিটন ভাইয়ের চায়ের দোকানে রাশেদ আজকের গল্প বলা শুরু করছে-

ডাবল ডাবলের বাহার

ধর, তুই দোকানে গিয়ে, চাল আর ডাল কিনে, পকেটে হাত দিয়ে দেখস পকেট খালি। একটা ফুটা পয়সাও নাই। এখন চাল আর ডালের দাম দেওয়ার জন্য বুদ্ধি বের করলি, তোর কোনো ফ্রেন্ডের কাছে টাকা ধার চাইবি। তবে যত টাকাই লাগুক না কেন, তুই ঠিক করলি ফ্রেন্ডের কাছে দ্বিগুণ টাকা ধার চাইবি। কারণ একবারে বেশি টাকা ধার নিতে পারলে, বারবার ধার চাওয়া লাগবে না।

যেহেতু দ্বিগুণ টাকা ধার চাওয়া লাগবে, সেহেতু কোনো সংখ্যাকে দ্বিগুণ করার একটা function লেখ। এই ফাংশনের নাম দে দুই গুণ কর (duiGoonKor)। কোনো একটা সংখ্যা দিয়ে এই ফাংশনকে কল করলে সেটার মান ডাবল করে return করে দিবে। এই দ্বিগুণ কর ফাংশনটা নিচের মতো করে লেখ-

function duiGoonKor(songkha){
    var goonFol = songkha * 2;
    return goonFol;
}

এখন 25 দিয়ে duiGoonKor ফাংশনকে কল করলে আউটপুট হিসেবে 50 দিবে। duiGoonKor (25);

একটা ফাংশনকে কল করলে যে আউটপুট পাওয়া যায়, সেটা চাইলে কোনো একটা ভ্যারিয়েবলের মান হিসেবে সেট করতে পারবি। জাস্ট একটা ভ্যারিয়েবল ডিক্লেয়ার করে, ডানপাশে কোনো মান না বসিয়ে ফাংশনকে কল করে দিবি। যেমন, দুইগুণ (duiGoon) নামে একটা ভ্যারিয়েবল ডিক্লেয়ার করে নিচের মতো করে লিখলে, duiGoonKor ফাংশনের আউটপুট duiGoon ভ্যারিয়েবলের মান হয়ে যাবে। var duiGoon = duiGoonKor (25);

বুদ্ধির সার ও প্রোগ্রামের ধার

চাল আর ডাল কিনতে মোট যত টাকা লাগে, তার দ্বিগুণ টাকা ধার করার পুরো প্রোগ্রাম লিখতে চাইলে প্রথমেই পুরো প্রোগ্রামটা ছোট ছোট অংশে ভাগ করে নিতে হবে। ছোট ছোট অংশে ভাগ করে নিলে, তোকে পুরা জিনিসটা নিয়ে একসাথে চিন্তা করা লাগবে না; বরং ছোট একটা অংশ নিবি, সেটার কাজ হয়ে গেলে তার পরবর্তী ছোট অংশ নিয়ে কাজ শুরু করবি। এখন দ্বিগুণ টাকা ধার করার পুরো প্রোগ্রামটাকে নিচের মতো করে ছয়টা ছোট ছোট অংশে ভাগ করবি-


  • ১. প্রথম অংশে চাল আর ডালের দাম যোগ করার জন্য যোগ কর (joogKor) ফাংশন লিখবি।

  • ২. কোনো একটা সংখ্যাকে দুইগুণ করার জন্য দুই গুণ কর (duiGoonKor) ফাংশন লিখবি।

  • ৩. ধার চাওয়ার জন্য ধার দে (dharDey) নামে একটা ফাংশন লিখবি। যেখানে ইনপুট হিসেবে নিবি ধারের পরিমাণ (dharAmount)। তারপর এই ফাংশনের ভিতরে ফ্রেন্ডের টাকা (frienderTaka) নামে একটা ভ্যারিয়েবল থাকবে। এই ভ্যারিয়েবলের মান হবে তুই যত টাকা ধার চাইবি তার সমান। পরের লাইনে frienderTaka ভ্যারিয়েবলকে রিটার্ন করে দিবি। 

    • function dharDey(dharAmount){
    var frienderTaka = dharAmount;
    return frienderTaka;
}

  • ৪. এখন চালের দাম এবং ডালের দাম নামে দুইটা ভ্যারিয়েবল ডিক্লেয়ার করে মান সেট করবি। এই দুইটা ভ্যারিয়েবলকে ইনপুট হিসেবে ব্যবহার করে যোগকর (joogKor) ফাংশনকে কল করবি। এই joogKor ফাংশন থেকে যে আউটপুট পাবি সেটা টোটাল (total) ভ্যারিয়েবলে রাখবি। তাহলে total ভ্যারিয়েবলের মান হবে চালের দাম আর ডালের দাম মিলে মোট যত টাকা হয়। 

    • var chalerDam = 28;
var dalerDam = 33;
var total = joogKor(chalerDam, dalerDam);

  • ৫. এতুই যেহেতু total টাকার দ্বিগুণ পরিমাণ ধার পাইতে চাস, সেহেতু total ভ্যারিয়েবলকে ইনপুট হিসেবে ব্যবহার করে দুই গুণ কর (duiGoonKor) ফাংশনকে কল করে, total ভ্যারিয়েবলকে দ্বিগুণ করে নে। এখন duiGoonKor ফাংশনের কল করলে যে আউটপুট পাওয়া যাবে, সেটাই হবে ধার লাগবে (dharLagbe) ভ্যারিয়েবলের মান।

    var dharLagbe = duiGoonKor(total);

  • ৬. সবশেষে dharLagbe ভ্যারিয়েবলকে ইনপুট হিসেবে ব্যবহার করে ধার দে (dharDey) ফাংশনকে কল কর। যে আউটপুট পাবি, সেটা টাকা পাইছি (takaPaichi) ভ্যারিয়েবলে রাখ। সবশেষে takaPaichi ভ্যারিয়েবলকে console.log করে দে। 

    var takaPaichi = dharDey(dharLagbe);
console.log(takaPaichi);

এখন দ্বিগুণ টাকা ধার পাওয়ার পুরা প্রোগ্রামটা একসাথে দেখ-

function joogKor(first, second){
 var joogFol = first + second;
 return joogFol;
}

function duiGoonKor(songkha){
    var goonFol = songkha * 2;
    return goonFol;
}

function dharDey(dharAmoount){
    var frienderTaka = dharAmoount;
    return frienderTaka;
}

var chalerDam = 28;
var dalerDam = 33;

var total = joogKor(chalerDam, dalerDam);
var dharLagbe = duiGoonKor(total);
var takaPaichi = dharDey(dharLagbe);

পুরো কোড www.jhankarmahbub.com/console-এ গিয়ে লিখলে আউটপুট হিসেবে 122 দেখাবে। দ্বিগুণ টাকা ধার চাওয়ার প্রোগ্রামটাকে ছয়টা অংশে ভাগ করার স্টেপগুলা কমপক্ষে তিনবার পড়বি। প্রথমবার পড়ে বুঝে ফেললেও তিনবার পড়বি। বড় প্রোগ্রামকে ছোট ছোট ভাগ করে প্রোগ্রামিং করার কনসেপ্ট ধরতে পারলে তুই অর্ধেক প্রোগ্রামার হয়ে যাবি। কারণ দুনিয়ার সব প্রোগ্রাম, সব সফটওয়্যারই ছোট ছোট অংশে ভাগ করে তৈরি করা হয়।

একের ভিতরে আরেক

একটা function-এর ভিতর থেকে আরেকটা function-কে কল করা যায়। চাইলে একটা ফাংশনের ভিতরে দুই-তিনটা ফাংশনকেও কল করতে পারবি। সেটা করার জন্য টাকা ধার দে (takaDharDey) নামে একটা function লেখ। সেটাতে first আর second ইনপুট হিসেবে নিবি। তারপর takaDharDey ফাংশনের ভিতর থেকেই joogKor, duiGoonKor, dharDey ফাংশনকে নিচের মতো করে কল করবি। সবশেষে takaPaichi ভ্যারিয়েবলকে আউটপুট হিসেবে রিটার্ন করে দে। 

function takaDorkar(first, second){
    var total = joogKor(first, second);
    var dharLagbe = duiGoonKor(total);
    var takaPaichi = dharDey(dharLagbe);
    return takaPaichi;
}

এখন তুই takarDorkar (28, 33) লিখে কল করলে 122 আউটপুট দিবে।

কত সেকেন্ডে কত ঘণ্টা

একটা ফাংশনের ভিতরে চাইলে একাধিক কাজও করতে পারস। ধর, তুই একটা ফাংশন লিখতে চাস। যেটাতে ইনপুট হিসেবে নিবি সেকেন্ড (seconds) তারপর সে বলে দিবে কত ঘণ্টা হয়। এই ফাংশনের নাম দিলি সেকেন্ড টু ঘণ্টা (secondsToHours)|

এই ফাংশনের ভিতরে প্রথমেই মিনিট (minutes) নামে একটা ভ্যারিয়েবল ডিক্লেয়ার করবি যার মান হবে সেকেন্ডকে 60 দিয়ে ভাগ করলে ভাগফল যত হয়। কারণ সেকেন্ডকে 60 দিয়ে ভাগ করলে মিনিট পাওয়া যায়। তার পরের লাইনে মিনিটকে (minutes) ঘণ্টায় রূপান্তর করবি। মিনিটকে ঘণ্টায় (hours) রূপান্তর করতে হলে কত দিয়ে ভাগ করতে হয়, আমি তোকে বলে দিব না। তোর আম্মুকে জিজ্ঞেস কর। সবশেষে যত ঘণ্টা হবে, সেটা আউটপুট হিসেবে return করে দে। ফাংশনটা নিচের মতো হবে- 

function secondsToHours (seconds){
 var minutes = seconds / 60;
 var hours = minutes / 60;
 return hours;
}

এখন secondToHours ফাংশনকে 7200 ইনপুট দিয়ে কল করলে আউটপুট হিসেবে 2 দিবে। কারণ 7200 সেকেন্ডে দুই ঘণ্টা হয়।
