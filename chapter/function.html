function বুঝবে না- কোনটা বউ কোনটা শালি

লিটন ভাইয়ের দোকানের সামনে এক ঝালমুড়িওয়ালাকে দেখে রাশেদ বলতে শুরু করল-

দশ টাকার ঝালমুড়ি

ঝালমুড়ি বানানোর জন্য মুড়ি, চানাচুর, পেঁয়াজ, মরিচ, লেবু, হাবিজাবি অনেক কিছু লাগে। যেসব জিনিস দিয়ে ঝালমুড়ি বানানো হয় তাদের বলতে পারস ঝালমুড়ি বানানোর উপাদান বা ইনপুট। এই উপাদানগুলা একটা প্লণাস্টিকের মগের মধ্যে নিয়ে, কাঠি দিয়ে ভালো করে নেড়ে নেড়ে ঝালমুড়ি বানায়। আর ঝালমুড়ি বানানোর কাজ শেষ হলে আউটপুট হিসেবে পাওয়া যায় এক প্যাকেট ঝালমুড়ি।

এখন তুই যদি কোনো ঝালমুড়িওয়ালাকে বলস— ভাই, 10 টাকার ঝালমুড়ি দিন। তাহলে সে ঝালমুড়ি বানানোর কাজ করে তোকে এক প্যাকেট ঝালমুড়ি দিবে। আবার তুই যদি বলস- ভাই, 20 টাকার ঝালমুড়ি দেন। তাহলে সে একই সিস্টেমে ঝালমুড়ি বানাবে। তবে 20 টাকার বললে বেশি বেশি মুড়ি, চানাচুর নিবে আর 10 টাকার বললে কম কম নিবে। তবে ঝালমুড়ি বানানোর সিস্টেম কিন্তু একই থাকবে। লিটন ভাইয়ের চা বানানো যেমন একটা প্রোগ্রাম, ঝালমুড়ি বানানোও একটা প্রোগ্রাম।

ফাংশন কী জিনিস

সব প্রোগ্রামের মধ্যেই ছোটখাটো খুচরা-খাচরা কাজ থাকে। ঝালমুড়ি বানানোর মধ্যেও অনেক খুচরা-খাচরা কাজ আছে। যেমন, মুড়ি নেওয়া, পেঁয়াজ কাটা, লেবু চিপা ইত্যাদি। বড় প্রোগ্রামের মধ্যে এইসব খুচরা-খাচরা কাজকে প্রোগ্রামিংয়ের ভাষায় function বলে। function একটা ইংরেজি শব্দ। এই function শব্দটাকে বাংলায় বলে কাজ। আর অনেকগুলা function একসাথ করেই একটা প্রোগ্রাম বানানো হয়।

ফাংশনলেখার সিস্টেম

সব function-এর একটা নাম দিতে হয়। যাতে সেই নাম ধরে ফাংশনটাকে ডাক দিতে পারস। যেমন, চানাচুর নেওয়ার function-এর নাম দিবি-‘চানাচুর দে’, যাতে এই নাম ধরে ফাংশনটাকে ডাক দিতে পারস। তা ছাড়া ‘চানাচুর দে’ নাম শুনলেই বুঝা যায় এই ফাংশনকে ডাক দিলে ফাংশনটা চানাচুর দিবে। একইভাবে পেঁয়াজ কাটার function হলে, সেটার নাম দিবি ‘পেঁয়াজ কাট’। যাতে নাম শুনলে বুঝা যায় ফাংশনটা কী কাজ করবে।

কোনো একটা ভ্যারিয়েবল ডিক্লেয়ার করার সময় ভ্যারিয়েবলের নামের আগে var লিখতে হয়। তেমনি কোনো একটা ফাংশন ডিক্লেয়ার করার সময় ফাংশনের নামের আগে function শব্দটা লিখতে হয়। যাতে function শব্দটা দেখলে বুঝতে পারে, তুই একটা function লিখতেছস। function শব্দ লিখার পর, ফাংশনের নাম লিখবি। তারপর function-এর মধ্যে যেসব ইনপুট নেওয়া হয়, সেসব ইনপুটগুলা প্রথম ব্র্যাকেটের ভিতরে লিখতে হবে। একাধিক ইনপুট থাকলে যেকোনো দুইটা ইনপুটের মাঝখানে কমা দেওয়া লাগবে। 

function name (input1, input2, input3){
    return output;
}

ইনপুট লিখা শেষ হয়ে গেলে সেকেন্ড ব্র্যাকেট দিতে হবে। কোনো একটা ফাংশনের ভিতরে যত কাজ হয় তার সবকিছু এই সেকেন্ড ব্র্যাকেটের ভিতরে হয়। ফাংশনের ভিতরের সব কাজ শেষ হলে যে আউটপুট দিতে হবে, সেটা return শব্দের পরে লিখতে হবে। return একটা ইংরেজি শব্দ। এইটার মানে ফেরত দেওয়া বা রিটার্ন করা। তাহলে ফাংশন কল করা হলে return-এর পরে যা লেখা থাকবে, সেটার মান রিটার্ন করবে বা আউটপুট হিসেবে দিয়ে দিবে।

ফাংশন লিখে যোগ করি

ধর, তোকে 10 আর 20 যোগ করতে বললাম। তুই ইনপুট হিসেবে 10 এবং 20-কে নিবি। তারপর যোগ করবি। যোগ করা শেষ হলে যোগফল আউটপুট হিসেবে বলে দিবি। এখন তোকে 15 আর 35-কে যোগ করতে বললে, তুই একই পদ্ধতিতে ইনপুট হিসেবে 15 এবং 35-কে নিয়ে, যোগ করে, যোগফল আউটপুট হিসেবে বলে দিবি। এই যোগ করার কাজটা যাতে সহজেই বারবার করতে পারস, সে জন্য তুই একটা ফাংশন লিখবি।

যেহেতু যোগ করার কাজ করবে এমন ফাংশন লিখতে চাস, সেহেতু এই ফাংশনের নাম দিবি যোগ কর (joogKor)। শুরুতে function শব্দ লিখে তারপর joogKor লিখবি। এই ফাংশনের ভিতরে দুইটা সংখ্যা যোগ করবি, তাই ইনপুট হিসেবে দুইটা সংখ্যা নিবি। joogKor লেখার পরে প্রথম ব্র্যাকেট দিয়ে ব্র্যাকেটের ভিতরে লিখবি firstNumber এবং কমা দিয়ে দ্বিতীয় ইনপুটের নাম লিখবি secondNumber। তারপর সেকেন্ড ব্র্যাকেট দিবি। এখন এই দুইটা সংখ্যাকে যোগ করার জন্য ফাংশনের সেকেন্ড ব্র্যাকেটের ভিতরে প্রথমেই একটা ভ্যারিয়েবল ডিক্লেয়ার করবি, যেটার নাম দিবি যোগ ফল (joogFol)। এই যোগফল ভ্যারিয়েবলের মান হবে firstNumber এবং secondNumber যোগ করলে যোগফল যত হয়। সে জন্যই var joogFol লিখে সমান চিহ্ন দিয়ে লিখবি firstNumber + secondNumber;

যোগ করা হয়ে গেলেই কিন্তু কাজ শেষ না। যোগফল আউটপুট হিসেবে বলে দিতে হবে। তুই যেহেতু joogFol কত হইছে বলে দিতে চাস, সেহেতু return joogFol; লিখবি। নিচের মতো করে- 

function joogKor(firstNumber, secondNumber){
    var joogFol = firstNumber + secondNumber;
    return joogFol;
}

ফাংশন ফাংশন ডাক পাড়ি

কোনো একটা function ব্যবহার করা খুবই সোজা। জাস্ট function-এর নাম লিখে ইনপুটগুলাকে প্রথম ব্র্যাকেটের ভিতরে কমা দিয়ে দিয়ে লিখতে হবে। ধর, তুই যদি 10 এবং 20 যোগ করতে চাস, তাহলে joogKor নামক function-টাকে নিচের মতো করে ডাক দিবি। কোনো function-কে ডাক দেওয়ার কাজকে প্রোগ্রামিংয়ের ভাষায় function-কে কল করা বলে। 

joogKor(10, 20);

ওপরের মতো করে joogKor নামক function-কে কল করলে joogKor-এর ভিতরে firstNumber-এর মান 10 হবে। কারণ joogKor ফাংশনকে কল করার সময় প্রথমেই 10 দিছস। secondNumber-এর মান 20 হবে। কারণ joogKor ফাংশনকে কল করার সময় সেকেন্ড পজিশনে 20 দিছস। এখন function-এর সেকেন্ড ব্র্যাকেটের ভিতরে, প্রথম লাইনের ডান পাশে আছে firstNumber + secondNumber । firstNumber-এর মান 10 আর secondNumber-এর মান 20। তাই firstNumber + secondNumber -এর মান হবে 30। সে জন্য যোগফল (joogFol) ভ্যারিয়েবলের মান হবে 30। তারপর function-এর ভিতরের সেকেন্ড লাইনে return joogFol; লেখা থাকায় joogFol ভ্যারিয়েবলের মান আউটপুট হিসেবে return করবে। যেহেতু joogFol ভ্যারিয়েবলের মান 30, সেহেতু আউটপুট হিসেবে 30 পাওয়া যাবে।

তুই চাইলে অন্য দুইটা সংখ্যাও যোগ করতে পারস। যেমন, 37 এবং 25 যোগ করতে চাইলে joogKor (37, 25); লিখে Run বাটনে চাপ দিবি। তখন আউটপুট দেখাবে 62 যা 37 ও 25 এর যোগফল। এ ক্ষেত্রে firstNumber-এর মান 37 আর secondNumber-এর মান 25। আবার joogKor (156, 213) লিখলে আউটপুট হিসেবে 369 দেখাবে।

ছোট ও সুন্দর নাম

তুই চাইলে firstNumber নামটা ছোট করে first আর secondNumber নামটা ছোট করে second লিখতে পারস। তবে ফার্স্ট ব্র্যাকেটের ভিতরে ইনপুটের যে নাম দিবি ফাংশনের ভিতরে সেই একই নাম ব্যবহার করতে হবে। আরেকটা কথা খেয়াল রাখবি, function কল করার সময় ব্র্যাকেটের ভিতরে প্রথমে যে মান দিবি, সেটা হবে প্রথম ইনপুট ভ্যারিয়েবলের মান। সেকেন্ডে যে মান দিবি, সেটা হবে সেকেন্ড ইনপুট ভ্যারিয়েবলের মান। 

function joogKor(first, second){
    var joogFol = first + second;
    return joogFol; 
}

ভ্যারিয়েবল ইনপুট

ফাংশন কল করার সময় ইনপুট হিসেবে ভ্যারিয়েবলও ব্যবহার করতে পারবি। ধর, তোর কাছে চালের দাম ও ডালের দাম নামে দুইটা ভ্যারিয়েবল আছে। 

var chalerDam = 28; 
var dalerDam = 20;

এখন চালের দাম আর ডালের দাম একত্রে কত হয়, সেটা জানার জন্য joogKor নামক function-কে নিচের মতো করে কল করতে পারবি। 

joogKor(chalerDam, dalerDam);

ওপরের কোড আউটপুট দেখাবে 48। কারণ চালের দাম ভ্যারিয়েবলের মান 28 টাকা আর ডালের দাম ভ্যারিয়েবলের মান 20 টাকা। এই দুইটা যোগ করলে যোগফল 48 টাকা হয়।

ফাংশন দিয়ে বিয়োগ করি

তুই যোগ করার ফাংশনের লিখার সিস্টেম বুঝতে পারলে, সহজেই বিয়োগ করার ফাংশন লিখতে পারবি। আগের মতো ফাংশনের নাম হিসেবে বিয়োগ কর (biyogKor) লিখবি। যাতে নাম শুনে বুঝা যায় function দিয়ে কী কাজ হবে। এখানেও ইনপুট ভ্যারিয়েবল হিসেবে first এবং second-কে রাখবি। তারপর, function-এর ভিতরে, আগে যেখানে যোগ করছিলি, সেখানে বিয়োগ করবি। সে জন্য বিয়োগফল (biyogFol) নামে একটা ভ্যারিয়েবল লিখবি। এই biyogFol ভ্যারিয়েবলের মান হবে first – second; অর্থাৎ দুইটা ইনপুট ভ্যারিয়েবলের বিয়োগফল। সবশেষে biyogFol ভ্যারিয়েবলকে রিটার্ন করে দে। তাহলেই কাজ শেষ। 

function biyogKor(first, second){
    var biyogFol = first - second;
    return biyogFol;
}

এখন 70 আর 50-এর বিয়োগফল বের করার জন্য biyogFol ফাংশনকে নিচের মতো কল করবি। তখন আউটপুট হিসেবে 20 দেখাবে। 

biyogKor (70, 50);

ধর, তুই সুন্দরবন ঘুরতে গিয়ে বনের মধ্যে পথ হারিয়ে রাতের অন্ধকারে বসে আছস। হুট করে তোর সামনে রয়েল বেঙ্গল টাইগার এসে হাজির। হালুম হালুম করতে করতে বলল “ফাংশন কী জিনিস, বল। না বলতে পারলে, তোকে খেয়ে ফেলব। তুই নাদুস-নুদুস আছস। খুব ভালো ডিনার হবে।” এই বলে খুশিতে লেজ নাড়াতে লাগল বাঘটা আর তুই ভয়ের চোটে কাঁপতে কাঁপতে বললি-

কোনো একটা প্রোগ্রামের খুচরা-খাচরা কাজকে function বলে। ফাংশনের একটা নাম থাকে, এক বা একাধিক ইনপুট নিতে পারে। একটা ফাংশনকে কল করে একই কাজ বারবার করা যায়।

এত কিছু কঠিন মনে হলে বলে দিবি- function মানে ঝালমুড়ি মিক্স করা।

প্রত্যেকটা প্রশ্নের উত্তর নিচে কমেন্ট কর। প্রথমে নিজে নিজে চেষ্টা কর। না পারলে অন্যদের কমেন্ট দেখে চেষ্টা কর। সেটাও করতে না পারলে, কেনো পরতেছস না, সেটা কমেন্ট কর।

১০.১: দুইটা সংখ্যা গুন করার একটা function লিখ। সেই ফাংশনে দুইটা ইনপুট নিবি তারপর সেই দুইটা ইনপুটকে গুন করে গুনফল একটা ভেরিয়েবলে রাখবি। শেষমেষ গুনফলটা return করে দিবি।

১০.২: তিনটা সংখ্যা যোগ করার একটা ফাংশন লিখ। যেহেতু তিনটা সংখ্যা যোগ করতে চাস, সেহেতু তিনটা ইনপুট ভেরিয়েবল নিবি। তারপর ফাংশনের ভিতরে সেই তিনটা ইনপুট ভেরিয়েবল যোগ করবি। বাকিটা তোর মাথা খাটিয়ে লিখে ফেল।
